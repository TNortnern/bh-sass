Basic HTML Integration

<!-- Embed the widget --> <iframe id="bh-widget" src="https://your-domain.com/embed/YOUR-TENANT/products?theme=light" width="100%" height="700" frameborder="0" allow="payment" ></iframe> <script> // Listen for widget events window.addEventListener('message', (event) => { if (!event.data?.type?.startsWith('bh:')) return; switch (event.data.type) { case 'bh:ready': console.log('Widget ready:', event.data); break; case 'bh:cart:updated': console.log('Cart:', event.data.cart, 'Total:', event.data.total); break; case 'bh:checkout:requested': console.log('Checkout URL:', event.data.checkoutUrl); break; } }); // Send commands to widget const widget = document.getElementById('bh-widget'); widget.contentWindow.postMessage({ type: 'bh:cart:add', itemId: '123' }, '*'); widget.contentWindow.postMessage({ type: 'bh:theme:set', theme: 'dark' }, '*'); </script>

WordPress (Custom HTML Block)

<!-- Add this to a Custom HTML block in WordPress --> <div id="bouncepro-widget-container" style="width: 100%; max-width: 1200px; margin: 0 auto;"> <iframe id="bouncepro-widget" src="https://your-domain.com/embed/YOUR-TENANT/products?theme=auto&hideFilters=false" style="width: 100%; min-height: 800px; border: none;" allow="payment" ></iframe> </div> <script> // Auto-resize iframe based on content window.addEventListener('message', function(e) { if (e.data?.type === 'bh:height:changed') { document.getElementById('bouncepro-widget').style.height = e.data.height + 'px'; } }); </script>

React Component

import { useEffect, useRef, useState } from 'react'; function BounceProWidget({ tenantSlug, theme = 'light', onCartUpdate }) { const iframeRef = useRef(null); const [height, setHeight] = useState(700); useEffect(() => { const handleMessage = (event) => { if (!event.data?.type?.startsWith('bh:')) return; switch (event.data.type) { case 'bh:cart:updated': onCartUpdate?.(event.data.cart, event.data.total); break; case 'bh:height:changed': setHeight(event.data.height); break; } }; window.addEventListener('message', handleMessage); return () => window.removeEventListener('message', handleMessage); }, [onCartUpdate]); const sendCommand = (type, payload = {}) => { iframeRef.current?.contentWindow?.postMessage({ type, ...payload }, '*'); }; return ( <iframe ref={iframeRef} src={`https://your-domain.com/embed/${tenantSlug}/products?theme=${theme}`} style={{ width: '100%', height, border: 'none' }} allow="payment" /> ); } // Usage: // <BounceProWidget // tenantSlug="my-business" // theme="dark" // onCartUpdate={(cart, total) => console.log(cart, total)} // />

Vue 3 Component

<template> <iframe ref="widgetRef" :src="widgetUrl" :style="{ width: '100%', height: height + 'px', border: 'none' }" allow="payment" /> </template> <script setup> import { ref, computed, onMounted, onUnmounted } from 'vue'; const props = defineProps({ tenantSlug: { type: String, required: true }, theme: { type: String, default: 'light' } }); const emit = defineEmits(['cart-update', 'checkout']); const widgetRef = ref(null); const height = ref(700); const widgetUrl = computed(() => `https://your-domain.com/embed/${props.tenantSlug}/products?theme=${props.theme}` ); function handleMessage(event) { if (!event.data?.type?.startsWith('bh:')) return; if (event.data.type === 'bh:cart:updated') { emit('cart-update', event.data.cart, event.data.total); } else if (event.data.type === 'bh:height:changed') { height.value = event.data.height; } else if (event.data.type === 'bh:checkout:requested') { emit('checkout', event.data); } } onMounted(() => window.addEventListener('message', handleMessage)); onUnmounted(() => window.removeEventListener('message', handleMessage)); function sendCommand(type, payload = {}) { widgetRef.value?.contentWindow?.postMessage({ type, ...payload }, '*'); } defineExpose({ sendCommand }); </script>

PostMessage API Reference

Events (Widget → Parent)

EventData
bh:ready{ tenantSlug, widgetType, version }
bh:cart:updated{ cart, total, itemCount }
bh:checkout:requested{ tenantSlug, cart, total, checkoutUrl }
bh:height:changed{ height }
bh:item:clicked{ item: { id, name, slug } }

Commands (Parent → Widget)

CommandPayload
bh:cart:add{ itemId, quantity? }
bh:cart:remove{ itemId }
bh:cart:clear(none)
bh:navigate{ to: 'checkout' | 'products' }
bh:theme:set{ theme: 'light' | 'dark' | 'auto' }

Widget URL Parameters

ParameterDefaultDescription
themeautolight, dark, or auto (follows system)
category-Filter by category ID
featuredfalseShow only featured items
limit12Items per page
hideFiltersfalseHide search/filter controls
hideCartfalseHide cart sidebar
